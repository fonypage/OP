#include <iostream>
using namespace std;
int main()
{
    setlocale(LC_ALL, "Russian");
    int n, ch = 0;
    cout << "Введите числовую последовательность оканчивающаяся на 0 : ";
    cin >> n;
    for (n; n != 0; cin >> n) // Ввод последовательности
    {
        while (n) // с 11 до 18 строки проверка тройки как элемента и цифры элемента
        {
            if (n % 10 == 3)
            {
                ch++;
            }
            n /= 10;
        }
    }
    cout << ch << " раз(а) тройка встретилась в последовательности";
}
